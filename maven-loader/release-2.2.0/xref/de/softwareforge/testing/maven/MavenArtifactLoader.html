64  /**
65   * A quick and dirty artifact loader. Downloads published artifacts from the Maven repository system.
66   * <p>
67   * The loader respects the local maven settings (repositories, mirrors etc.) if present. If no configuration is found, a hard-coded reference to <a
68   * href="https://repo.maven.apache.org/maven2/">Maven Central</a> is used.
69   */
70  public final class MavenArtifactLoader {
71  
72      private static final Logger LOG = LoggerFactory.getLogger(MavenArtifactLoader.class);
73  
74      private static final RemoteRepository CENTRAL_REPO = new RemoteRepository.Builder("central", "default", "https://repo.maven.apache.org/maven2/").build();
75  
76      private static final String USER_HOME = System.getProperty("user.home");
77      private static final File USER_MAVEN_HOME = new File(USER_HOME, ".m2");
78      private static final String ENV_M2_HOME = System.getenv("M2_HOME");
79  
80      private static final File DEFAULT_USER_SETTINGS_FILE = new File(USER_MAVEN_HOME, "settings.xml");
81      private static final File DEFAULT_USER_REPOSITORY = new File(USER_MAVEN_HOME, "repository");
82      private static final File DEFAULT_GLOBAL_SETTINGS_FILE =
83              new File(System.getProperty("maven.home", Objects.requireNonNullElse(ENV_M2_HOME, "")), "conf/settings.xml");
84  
85      private final RepositorySystem repositorySystem;
86      private final RepositorySystemSession mavenSession;
87      private final List<RemoteRepository> remoteRepositories;
88  
89      private final String extension;

91      /**
92       * Creates a new artifact loader for 'jar' artifacts.
93       */
94      public MavenArtifactLoader() {
95          this("jar");
96      }
97  
98      /**
99       * Creates a new artifact loader for artifacts.
100      *
101      * @param extension The artifact extension. Must not be null.
102      */
103     public MavenArtifactLoader(String extension) {
104         this.extension = requireNonNull(extension, "extension is null");
105 
106         @SuppressWarnings("deprecation")
107         ServiceLocator serviceLocator = createServiceLocator();
108         this.repositorySystem = serviceLocator.getService(RepositorySystem.class);
109 
110         try {
111             Settings settings = createSettings();
112             File localRepositoryLocation = settings.getLocalRepository() != null ? new File(settings.getLocalRepository()) : DEFAULT_USER_REPOSITORY;
113             LocalRepository localRepository = new LocalRepository(localRepositoryLocation);
114             this.remoteRepositories = extractRemoteRepositories(settings);
115 
116             DefaultRepositorySystemSession mavenSession = MavenRepositorySystemUtils.newSession();
117 
118             this.mavenSession = mavenSession.setLocalRepositoryManager(repositorySystem.newLocalRepositoryManager(mavenSession, localRepository));
119 
120         } catch (SettingsBuildingException e) {
121             throw new IllegalStateException("Could not load maven settings:", e);
122         }
123     }

125     /**
126      * Create a new version match builder to retrieve an artifact.
127      *
128      * @param groupId    The Apache Maven Group Id. Must not be null.
129      * @param artifactId The Apache Maven Artifact Id. Must not be null.
130      * @return A {@link MavenVersionMatchBuilder} instance
131      */
132     public MavenVersionMatchBuilder builder(String groupId, String artifactId) {
133         requireNonNull(groupId, "groupId is null");
134         requireNonNull(artifactId, "artifactId is null");
135 
136         return new MavenVersionMatchBuilder(this, groupId, artifactId);
137     }

139     /**
140      * Download an artifact file from the Maven repository system.
141      *
142      * @param groupId    The Apache Maven Group Id. Must not be null.
143      * @param artifactId The Apache Maven Artifact Id. Must not be null.
144      * @param version    The Apache Maven Artifact version. Must not be null.
145      * @return A file representing a successfully downloaded artifact.
146      * @throws IOException If the artifact could not be found or an IO problem happened while locating or downloading the artifact.
147      */
148     public File getArtifactFile(String groupId, String artifactId, String version) throws IOException {
149         requireNonNull(groupId, "groupId is null");
150         requireNonNull(artifactId, "artifactId is null");
151         requireNonNull(version, "version is null");
152 
153         ArtifactRequest artifactRequest = new ArtifactRequest();
154         artifactRequest.setArtifact(new DefaultArtifact(groupId, artifactId, extension, version));
155         artifactRequest.setRepositories(this.remoteRepositories);
156         try {
157             ArtifactResult artifactResult = this.repositorySystem.resolveArtifact(mavenSession, artifactRequest);
158             Artifact artifact = artifactResult.getArtifact();
159             return artifact.getFile();
160         } catch (RepositoryException e) {
161             throw new IOException(e);
162         }
163     }
164 
165     /**
166      * Find a matching artifact version from a partially defined artifact version.
167      * <p>
168      * Any located artifact in the repository system is compared to the version given.
169      * <br>
170      * Using this method is equivalent to calling
171      * <pre>
172      *  builder(groupId, artifactId)
173      *                 .partialMatch(version)
174      *                 .extension(extension)
175      *                 .includeSnapshots(true)
176      *                 .findBestMatch();
177      * </pre>
178      *
179      * but will throw an IOException if no version could be found.
180      *
181      * @param groupId    The Apache Maven Group Id. Must not be null.
182      * @param artifactId The Apache Maven Artifact Id. Must not be null.
183      * @param version    A partial version string. Must not be null. An empty string matches any version.
184      * @return The latest version that matches the partial version string. It either starts with the partial version string given (an empty version string
185      * matches any version) or is exactly the provided version.
186      * @throws IOException If an IO problem happened during artifact download or no versions were found during resolution.
187      */
188     public String findLatestVersion(String groupId, String artifactId, String version) throws IOException {
189         requireNonNull(groupId, "groupId is null");
190         requireNonNull(artifactId, "artifactId is null");
191         requireNonNull(version, "version is null");
192         return builder(groupId, artifactId)
193                 .partialMatch(version)
194                 .extension(extension)
195                 .includeSnapshots(true)
196                 .findBestMatch()
197                 .orElseThrow(() -> new IOException(format("No suitable candidate for %s:%s:%s found!", groupId, artifactId, version)));
198     }

200     SortedSet<Version> findAllVersions(MavenVersionMatchBuilder builder) throws IOException {
201 
202         Artifact artifact = new DefaultArtifact(builder.groupId(), builder.artifactId(), builder.extension(), "[0,)");
203 
204         VersionRangeRequest rangeRequest = new VersionRangeRequest();
205         rangeRequest.setArtifact(artifact);
206         rangeRequest.setRepositories(this.remoteRepositories);
207 
208         try {
209             VersionRangeResult rangeResult = this.repositorySystem.resolveVersionRange(mavenSession, rangeRequest);
210             SortedSet<Version> resultBuilder = new TreeSet<>();
211             List<Version> artifactVersions = rangeResult.getVersions();
212             VersionStrategy versionStrategy = builder.versionStrategy();
213             if (artifactVersions != null) {
214                 for (Version artifactVersion : artifactVersions) {
215                     boolean isSnapshot = artifactVersion.toString().endsWith("-SNAPSHOT");
216                     boolean match = versionStrategy.matchVersion(artifactVersion);
217 
218                     // remove match if snapshots are not requested but the version is a snapshot
219                     if (isSnapshot) {
220                         match &= builder.includeSnapshots();
221                     }
222 
223                     if (match) {
224                         resultBuilder.add(artifactVersion);
225                     }
226                 }
227             }
228             return Collections.unmodifiableSortedSet(resultBuilder);
229         } catch (VersionRangeResolutionException e) {
230             throw new IOException(format("Could not resolve version range: %s", rangeRequest), e);
231         }
232     }

235     @SuppressWarnings("deprecation")
236     private static ServiceLocator createServiceLocator() {
237         DefaultServiceLocator locator = MavenRepositorySystemUtils.newServiceLocator();
238 
239         locator.addService(RepositoryConnectorFactory.class, BasicRepositoryConnectorFactory.class);
240         locator.addService(TransporterFactory.class, FileTransporterFactory.class);
241         locator.addService(TransporterFactory.class, HttpTransporterFactory.class);
242 
243         locator.setErrorHandler(new DefaultServiceLocator.ErrorHandler() {
244             @Override
245             public void serviceCreationFailed(Class<?> type, Class<?> impl, Throwable e) {
246                 LOG.error(format("Could not create instance of %s (implementation %s): ", type.getSimpleName(), impl.getSimpleName()), e);
247             }
248         });
249 
250         return locator;
251     }

253     private static Settings createSettings() throws SettingsBuildingException {
254         SettingsBuildingRequest settingsBuildingRequest = new DefaultSettingsBuildingRequest()
255                 .setSystemProperties(System.getProperties())
256                 .setUserSettingsFile(DEFAULT_USER_SETTINGS_FILE)
257                 .setGlobalSettingsFile(DEFAULT_GLOBAL_SETTINGS_FILE);
258 
259         DefaultSettingsBuilderFactory settingBuilderFactory = new DefaultSettingsBuilderFactory();
260         DefaultSettingsBuilder settingsBuilder = settingBuilderFactory.newInstance();
261         SettingsBuildingResult settingsBuildingResult = settingsBuilder.build(settingsBuildingRequest);
262 
263         return settingsBuildingResult.getEffectiveSettings();
264     }
265 
266     private static List<RemoteRepository> extractRemoteRepositories(Settings settings) {
267         Map<String, Profile> profiles = settings.getProfilesAsMap();
268         List<RemoteRepository> builder = new ArrayList<>();
269 
270         boolean foundRepository = false;
271         for (String profileName : settings.getActiveProfiles()) {
272             Profile profile = profiles.get(profileName);
273             if (profile != null) {
274                 List<Repository> repositories = profile.getRepositories();
275                 if (repositories != null) {
276                     for (Repository repo : repositories) {
277                         builder.add(new RemoteRepository.Builder(repo.getId(), "default", repo.getUrl()).build());
278                         foundRepository = true;
279                     }
280                 }
281             }
282         }
283 
284         if (!foundRepository && !settings.isOffline()) {
285             builder.add(CENTRAL_REPO);
286         }
287 
288         return Collections.unmodifiableList(builder);
289     }
